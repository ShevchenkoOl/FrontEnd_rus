                id: Date.now(),   // –≤ –¥–∞–Ω–Ω–æ–º –ø—Ä–∏–º–µ—Ä–µ ID —Å–æ–∑–¥–∞—ë—Ç—å—Å—è –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ —Å –ø–æ–º–æ—â—å—é —Å–µ—Ä–≤–µ—Ä–∞ json-server, –Ω–æ –¥–ª—è –±—É–¥—É—â–µ–≥–æ –ª—É—á—à–µ —Å–æ–∑–¥–∞–≤–∞—Ç—å  —Å–∞–º–æ–º—É
                author: author,
                content: content,
                likes: 0 // –Ω–∞—á–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –ª–∞–π–∫–æ–≤
            });
        }

        // –û—á–∏—â–∞–µ–º –ø–æ–ª—è —Ñ–æ—Ä–º—ã
        authorInput.value = "";
        contentInput.value = "";

        // –ü–µ—Ä–µ–∑–∞–≥—Ä—É–∂–∞–µ–º —Å–ø–∏—Å–æ–∫ –ø–æ—Å—Ç–æ–≤
        loadPosts();

    } catch (error) {
        // –í—ã–≤–æ–¥ –æ—à–∏–±–∫–∏ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è
        console.error("–û—à–∏–±–∫–∞ –ø—Ä–∏ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏–∏ –ø–æ—Å—Ç–∞", error);
    }
});


// ============================
// –£–¥–∞–ª–µ–Ω–∏–µ –ø–æ—Å—Ç–∞ (DELETE)
// ============================

// –ê—Å–∏–Ω—Ö—Ä–æ–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è —É–¥–∞–ª–µ–Ω–∏—è –ø–æ—Å—Ç–∞
async function deletePost(id) {
    try {
        // DELETE-–∑–∞–ø—Ä–æ—Å –ø–æ id –ø–æ—Å—Ç–∞
        await axios.delete(`http://localhost:3001/posts/${id}`);

        // –û–±–Ω–æ–≤–ª—è–µ–º —Å–ø–∏—Å–æ–∫
        loadPosts();
    } catch (error) {
        console.error("–û—à–∏–±–∫–∞ –ø—Ä–∏ —É–¥–∞–ª–µ–Ω–∏–∏ –ø–æ—Å—Ç–∞", error);
    }
}

// –ê—Å–∏–Ω—Ö—Ä–æ–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –ª–∞–π–∫–∞
async function likePost(post) {
    try {
        // PATCH-–∑–∞–ø—Ä–æ—Å ‚Äî —á–∞—Å—Ç–∏—á–Ω–æ–µ –æ–±–Ω–æ–≤–ª–µ–Ω–∏–µ
        await axios.patch(`http://localhost:3001/posts/${post.id}`, {
            likes: post.likes + 1
        });

        // –û–±–Ω–æ–≤–ª—è–µ–º —Å–ø–∏—Å–æ–∫
        loadPosts();
    } catch (error) {
        console.error("–û—à–∏–±–∫–∞ –ø—Ä–∏ –ª–∞–π–∫–µ –ø–æ—Å—Ç–∞", error);
    }
}

function openEditForm(post) {

    authorInput.value = post.author;

    // –ó–∞–ø–∏—Å—ã–≤–∞–µ–º —Ç–µ–∫—Å—Ç –ø–æ—Å—Ç–∞
    contentInput.value = post.content;

    submitBtn.textContent = "–°–æ—Ö—Ä–∞–Ω–∏—Ç—å –∏–∑–º–µ–Ω–µ–Ω–∏—è";

    editPostId = post.id;
}
